[
    {
        "前置条件": ["注册界面正常可访问"],
        "操作步骤": ["用户打开注册页面"],
        "预期结果": ["系统成功创建一个新用户"],

        "操作步骤": ["用户输入有效的用户名、密码和电子邮件地址"],

        "前置条件": ["系统支持不同浏览器，操作系统和设备"],
